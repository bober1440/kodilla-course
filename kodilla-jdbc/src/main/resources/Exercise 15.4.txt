SELECT B.NAME AS LISTNAME, A.SUMMARY AS ISSUE
FROM ISSUES A
JOIN ISSUESLISTS B ON A.ISSUESLIST_ID = B.ID
ORDER BY B.ID;

SELECT U.FIRSTNAME, U.LASTNAME, A.SUMMARY AS ISSUE
FROM ISSUES A
JOIN USERS U on A.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.FIRSTNAME,U.LASTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_COUNTER
FROM ISSUES A
JOIN USERS U ON USER_ID_ASSIGNEDTO = U.ID
GROUP BY A.USER_ID_ASSIGNEDTO
ORDER BY U.FIRSTNAME, U.LASTNAME;
